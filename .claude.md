```typescript
// app/api/experiences/route.ts
import { createClient } from '@/lib/supabase/server'
import { NextResponse } from 'next/server'
import { z } from 'zod'

const CreateExperienceSchema = z.object({
  story_text: z.string().optional(),
  category: z.string(),
  tags: z.array(z.string()),
  // ... exact schema from API-SPECIFICATION.md
})

export async function POST(req: Request) {
  const supabase = createClient()

  // 1. Auth check
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return NextResponse.json(
      { error: 'Unauthorized', code: 'UNAUTHORIZED' },
      { status: 401 }
    )
  }

  // 2. Validate input
  const body = await req.json()
  const validated = CreateExperienceSchema.safeParse(body)
  if (!validated.success) {
    return NextResponse.json(
      { error: 'Invalid request', code: 'VALIDATION_ERROR', details: validated.error },
      { status: 400 }
    )
  }

  // 3. Insert (RLS enforced!)
  const { data, error } = await supabase
    .from('experiences')
    .insert({ ...validated.data, user_id: user.id })
    .select()
    .single()

  if (error) {
    return NextResponse.json(
      { error: error.message, code: 'DATABASE_ERROR' },
      { status: 500 }
    )
  }

  // 4. Return exact schema
  return NextResponse.json(data, { status: 201 })
}
```

// app/api/ai/analyze-text/route.ts
import { NextResponse } from 'next/server'
import OpenAI from 'openai'

const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY })

export async function POST(req: Request) {
  const { text, language = 'de' } = await req.json()

  try {
    const completion = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      messages: [
        {
          role: 'system',
          content: `Analyze this experience and extract: category, tags, emotions, location, date.
          Respond in JSON format.`
        },
        { role: 'user', content: text }
      ],
      response_format: { type: 'json_object' },
      timeout: 10000
    })

    const analysis = JSON.parse(completion.choices[0].message.content!)

    return NextResponse.json(analysis)
  } catch (error: any) {
    if (error.status === 429) {
      return NextResponse.json(
        { error: 'Rate limit exceeded', code: 'RATE_LIMIT_EXCEEDED' },
        { status: 429 }
      )
    }

    return NextResponse.json(
      { error: 'AI service error', code: 'AI_SERVICE_ERROR' },
      { status: 503 }
    )
  }
}
```

'use client'

import { useState, useTransition } from 'react'
import { Button } from '@/components/ui/button'
import { Heart } from 'lucide-react'

interface LikeButtonProps {
  experienceId: string
  initialCount: number
}

export function LikeButton({ experienceId, initialCount }: LikeButtonProps) {
  const [count, setCount] = useState(initialCount)
  const [isLiked, setIsLiked] = useState(false)
  const [isPending, startTransition] = useTransition()

  const handleLike = () => {
    // Optimistic update
    setIsLiked(!isLiked)
    setCount(prev => isLiked ? prev - 1 : prev + 1)

    // Server action
    startTransition(async () => {
      await fetch(`/api/experiences/${experienceId}/upvote`, { method: 'POST' })
    })
  }

  return (
    <Button
      variant="ghost"
      size="sm"
      onClick={handleLike}
      disabled={isPending}
      aria-label={`Like experience (${count} likes)`}
      aria-pressed={isLiked}
    >
      <Heart className={cn("w-4 h-4 mr-2", isLiked && "fill-red-500 text-red-500")} />
      {count}
    </Button>
  )
}
```
