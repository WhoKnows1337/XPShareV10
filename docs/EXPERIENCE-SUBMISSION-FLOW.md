  yourExperience: Experience
  similarExperiences: Experience[]
  waveName: string
}

export function ClusterFormationAnimation({ yourExperience, similarExperiences, waveName }: Props) {
  const [scope, animate] = useAnimate()
  const [stage, setStage] = useState<'init' | 'attract' | 'connect' | 'wave'>('init')

  useEffect(() => {
    async function runAnimation() {
      // Stage 1: Your XP appears (0-1s)
      await animate('.your-xp', { scale: [0, 1.2, 1], opacity: [0, 1] }, { duration: 0.8 })

      // Stage 2: Similar XPs fly in (1-3s)
      setStage('attract')
      await animate(
        '.similar-xp',
        {
          x: [300, 0],
          y: [Math.random() * 200 - 100, 0],
          opacity: [0, 1],
          scale: [0, 1]
        },
        { duration: 0.6, delay: stagger(0.2) }
      )

      // Stage 3: Lines connect (3-4s)
      setStage('connect')
      await animate('.connection-line', { pathLength: [0, 1] }, { duration: 0.8 })

      // Stage 4: Wave-Badge appears (4-5s)
      setStage('wave')
      await animate('.wave-badge', {
        scale: [0, 1.3, 1],
        rotate: [0, 10, -10, 0]
      }, { duration: 1 })
    }

    runAnimation()
  }, [])

  return (
    <div ref={scope} className="relative h-[300px] flex items-center justify-center">
      {/* Your Experience (Center) */}
      <motion.div
        className="your-xp absolute z-10"
        style={{ left: '50%', top: '50%', translateX: '-50%', translateY: '-50%' }}

        <div className="w-16 h-16 rounded-full bg-primary flex items-center justify-center">
          <span className="text-2xl">â—</span>
        </div>
        <p className="text-xs mt-2 text-center font-semibold">Deine XP</p>
      </motion.div>

      {/* Similar Experiences (Circle around center) */}
      {similarExperiences.slice(0, 3).map((exp, i) => {
        const angle = (i * 120 * Math.PI) / 180 // 120Â° apart
        const radius = 100
        const x = Math.cos(angle) * radius
        const y = Math.sin(angle) * radius

        return (
          <motion.div
            key={exp.id}
            className="similar-xp absolute"
            style={{
              left: `calc(50% + ${x}px)`,
              top: `calc(50% + ${y}px)`,
              translateX: '-50%',
              translateY: '-50%'
            }}
          >
            <div className="w-12 h-12 rounded-full bg-primary/40 flex items-center justify-center">
              <span className="text-lg">â—‹</span>
            </div>
          </motion.div>
        )
      })}

      {/* Connection Lines */}
      {stage !== 'init' && (
        <svg className="absolute inset-0 w-full h-full" style={{ zIndex: 1 }}>
          {similarExperiences.slice(0, 3).map((exp, i) => {
            const angle = (i * 120 * Math.PI) / 180
            const radius = 100
            const x1 = 50 // center %
            const y1 = 50
            const x2 = 50 + (Math.cos(angle) * radius) / 3 // adjust for % coords
            const y2 = 50 + (Math.sin(angle) * radius) / 3

            return (
              <motion.line
                key={i}
                className="connection-line"
                x1={`${x1}%`}
                y1={`${y1}%`}
                x2={`${x2}%`}
                y2={`${y2}%`}

                strokeWidth="2"
                strokeDasharray="5,5"

MATCH (source:Experience {id: $experienceId})

RETURN inspired, r
ORDER BY r.similarity DESC

// hooks/useLiveAnalysis.ts

import { useEffect, useRef } from 'react'
import { useSubmissionStore } from '@/stores/submissionStore'
import { debounce } from 'lodash'

export function useLiveAnalysis() {
  const content = useSubmissionStore((state) => state.content)
  const setAnalysis = useSubmissionStore((state) => state.setAnalysis)

  const analysisCallCount = useRef(0)
  const MAX_CALLS = 10 // Kosten-Limit

  useEffect(() => {
    // Debounced AI-Call
    const analyzeContent = debounce(async () => {
      // Zu kurz?
      if (content.length < 50) return

      // Zu viele Calls?
      if (analysisCallCount.current >= MAX_CALLS) return

      analysisCallCount.current++

      try {
        const response = await fetch('/api/experiences/analyze', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ text: content })
        })

        const result = await response.json()

        // Update Store
        setAnalysis(result)

      } catch (error) {
        console.error('Analysis failed:', error)
      }
    }, 2000) // 2 Sekunden debounce

    analyzeContent()

    return () => analyzeContent.cancel()
  }, [content, setAnalysis])
}

      clearInterval(interval)
      window.removeEventListener('beforeunload', handleBeforeUnload)
    }
  }, [content, saveDraft])
}

// hooks/useDynamicQuestions.ts

import { useMemo } from 'react'
import { questionSets } from '@/data/questionSets'

export function useDynamicQuestions(
  category: string,
  extractedData: any
) {
  return useMemo(() => {
    const questions = questionSets[category] || questionSets.default

    // Filter: Nur Fragen, deren Info noch nicht extrahiert wurde
    const filteredQuestions = questions.filter(q => {
      // Wenn Zeit schon bekannt, keine Frage nach Dauer
      if (q.id === 'duration' && extractedData.time) return false

      // Wenn Ort schon bekannt, keine Frage nach location_type
      if (q.id === 'location_type' && extractedData.location) return false

      // Wenn Emotion schon erkannt, nicht nochmal fragen
      if (q.id === 'emotion' && extractedData.emotion) return false

      return true
    })

    // Sortiere nach PrioritÃ¤t
    return filteredQuestions
      .sort((a, b) => a.priority - b.priority)
      .slice(0, 5) // Max 5 Fragen

  }, [category, extractedData])
}

// lib/offline-queue.ts
import { openDB } from 'idb'

const dbPromise = openDB('xp-share-offline', 1, {
  upgrade(db) {
    db.createObjectStore('experiences', {
      keyPath: 'id',
      autoIncrement: true
    })
  }

export async function saveToOfflineQueue(data: any) {
  const db = await dbPromise
  await db.add('experiences', {
    ...data,
    offlineSavedAt: new Date().toISOString(),
    synced: false
  })
}

export async function getOfflineQueue() {
  const db = await dbPromise
  return db.getAll('experiences')
}

export async function syncOfflineQueue() {
  const db = await dbPromise
  const queue = await db.getAll('experiences')

  for (const item of queue) {
    try {
      await fetch('/api/experiences', {
        method: 'POST',
        body: JSON.stringify(item),
        headers: { 'Content-Type': 'application/json' }
      })

      await db.delete('experiences', item.id)
    } catch (err) {
      console.error('Sync failed:', err)
    }
  }
}
